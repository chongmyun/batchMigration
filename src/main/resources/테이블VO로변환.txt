SELECT
	'private' ||
	(CASE A.DATA_TYPE
		WHEN 'NUMBER'
		THEN 'long'
		WHEN 'DATE'
		THEN 'Date'
		WHEN 'TIMESTAMP'
		THEN 'LocalDateTime'
		WHEN 'VARCHAR2'
		THEN 'String'
		WHEN 'CHAR'
		THEN 'String'
		WHEN 'CLOB'
		THEN 'String'
		WHEN 'BLOB'
		THEN 'Byte[]'
		ELSE ''
	END
	) ||
	' '||
	LOWER(SUBSTR(A.CAMEL_CASE,1,1)) ||
	SUBSTR(A.CAMEL_CASE,2) ||
	';' AS RESULT
FROM(
	SELECT
		COLUMN_NAME,
		DATA_TYPE,
		REPLACE(INITCAP(COLUMN_NAME),'_') AS CAMEL_CASE
	FROM USER_TAB_COLUMNS
	WHERE TABLE_NAME=''
	ORDER BY COLUMN_ID
)A;
